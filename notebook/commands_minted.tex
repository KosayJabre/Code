\usepackage{minted}

\definecolor{tabcolor}{rgb}{0.85,0.85,0.85}

\newcommand{\inputcpp}[1]{\vspace{-3mm}\inputminted[
  fontsize=\footnotesize,
  breaklines=true,
  breakindent=0pt,
  breakbefore=./\,\_,
  breakautoindent=false,
  showtabs=true,
  tab=.,
  tabcolor=tabcolor,
  tabsize=1,
  stripall=true
]{cpp}{#1}\vspace{-5mm}}
