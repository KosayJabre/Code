\usepackage{minted}

\def \precodevspace {-1mm}
\def \postcodevspace {-3mm}
\definecolor{tabcolor}{rgb}{0.85,0.85,0.85}

\newcommand{\inputcpp}[1]{\vspace{\precodevspace}\inputminted[
  fontsize=\footnotesize,
  breaklines=true,
  breakindent=0pt,
  breakbefore=./\,\_,
  breakautoindent=false,
  showtabs=true,
  tab=.,
  tabcolor=tabcolor,
  tabsize=1,
  stripall=true,
  baselinestretch=0.825
]{cpp}{#1}\vspace{\postcodevspace}}
