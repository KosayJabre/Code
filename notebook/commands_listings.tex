\usepackage{listings}
\definecolor{dkgreen}{rgb}{0,0.6,0}
\definecolor{gray}{rgb}{0.5,0.5,0.5}
\definecolor{mauve}{rgb}{0.58,0,0.82}
\lstset{frame=none,
  language=C++,
  aboveskip=0mm,
  belowskip=0mm,
  showstringspaces=false,
  columns=flexible,
  basicstyle={\footnotesize\ttfamily},
  numbers=none,
  numberstyle=\tiny\color{gray},
  keywordstyle=\color{blue},
  commentstyle=\color{dkgreen},
  stringstyle=\color{mauve},
  breaklines=true,
  breakatwhitespace=false,
  breakindent=0pt,
  tabsize=1,
  postbreak=\mbox{\textcolor{red}{$\hookrightarrow$}\space}
}

\lstdefinestyle{StyleCPP}{language=C++}
\lstdefinestyle{StylePython}{language=Python}
\lstdefinestyle{StyleJava}{language=Java}
\lstdefinestyle{StyleBash}{language=Bash}

\newcommand{\inputcpp}[1]{\vspace{-1mm}\lstinputlisting[style=StyleCPP]{#1}}
\newcommand{\inputpython}[1]{\vspace{-1mm}\lstinputlisting[style=StylePython]{#1}}
\newcommand{\inputjava}[1]{\vspace{-1mm}\lstinputlisting[style=StyleJava]{#1}}
\newcommand{\inputbash}[1]{\vspace{-1mm}\lstinputlisting[style=StyleBash]{#1}}
